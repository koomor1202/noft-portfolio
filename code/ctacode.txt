<script>
  (function () {
    var pcCta = document.getElementById("noftPcCta");
    var spCta = document.getElementById("noftSpCta");

    function toggleCta() {
      var isSp = window.innerWidth <= 767;
      var isPc = window.innerWidth >= 768;
      var show = window.scrollY > 220;

      if (pcCta) {
        pcCta.classList.toggle("is-visible", isPc && show);
      }

      if (spCta) {
        spCta.classList.toggle("is-visible", isSp && show);
      }
    }

    window.addEventListener("scroll", toggleCta, { passive: true });
    window.addEventListener("resize", toggleCta);
    toggleCta();
  })();
</script>
